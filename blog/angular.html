# ant zooro 时间日期选择器-时间范围 0.7版本

把日期选择器修改为两个单独的日期选择器，进行时间绑定控制另个一个选择器的可选范围~

<nz-form-control>
  <!-- <nz-input-group>
    <span class="form-label">创建日期：</span><nz-range-picker [nzFormat]="dateFormat" formControlName="starttime"  (ngModelChange)="changeTimeFn($event)"></nz-range-picker>
  </nz-input-group> -->
  <nz-input-group>
    <span class="form-label">创建日期：</span><nz-date-picker
    formControlName="starttime"
    [nzAllowClear]="false"
    [nzDisabledDate]="disabledStartDateSearch"
    nzFormat="yyyy-MM-dd"
    [(ngModel)]="startTimeValueSearch"
    nzPlaceHolder="开始日期"
    (ngModelChange)="onStartChangeSearch($event)"
  ></nz-date-picker>-<nz-date-picker
  [nzDisabledDate]="disabledEndDateSearch"
  formControlName="endtime"
  [nzAllowClear]="false"
  nzFormat="yyyy-MM-dd"
  [(ngModel)]="endTimeValueSearch"
  nzPlaceHolder="结束日期"
  (ngModelChange)="onEndChangeSearch($event)"></nz-date-picker>
  </nz-input-group>
</nz-form-control>

disabledStartDateSearch = (startValue: Date): boolean => {
  if (!startValue || !this.endTimeValueSearch) {
    return false;
  }
  return ((startValue.getTime() < (this.endTimeValueSearch.getTime() - 60 * 24 * 60 * 60 * 1000 )) || (startValue.getTime() > this.endTimeValueSearch.getTime()));
}
disabledEndDateSearch = (endValue: Date): boolean => {
  if (!endValue || !this.startTimeValueSearch) {
    return false;
  }
  return ((endValue.getTime() > (this.startTimeValueSearch.getTime() + 60 * 24 * 60 * 60 * 1000 )) || (endValue.getTime() <= this.startTimeValueSearch.getTime()));
}
onStartChangeSearch(date: Date): void {
  console.log('这里有默认事件吗')
  this.startTimeValueSearch = date;
  this.changeTimeFn(date, 'search', 'start')
}
onEndChangeSearch(date: Date): void {
  this.endTimeValueSearch = date;
  this.changeTimeFn(date, 'search', 'end')
}
