# 深拷贝

 const isObj = obj => {
    return Object.prototype.toString.call(obj) === '[object Object]'
  }
  const isArray = obj => {
    return Array.isArray(obj)
  }
  const extend = (...args) => {
    let deep, target, sources
    target = args.shift()
    if (typeof target === 'boolean') {
      deep = target
      target = args.shift()
    }
    sources = args
    for (let index = 0; index < sources.length; index++) {
      let source = sources[index]
      for (let key in source) {
        if (deep && (isObj(source[key]) || isArray(source[key]))) {
          if (isObj(source[key]) && !isObj(target[key])) {
            target[key] = {}
          }
          if (isArray(source[key]) && !isArray(target[key])) {
            target[key] = []
          }
          extend(deep, target[key], source[key])
        } else if (source[key] !== undefined) {
          target[key] = source[key]
        }
      }
    }
    return target
  }

getParamStr (paramData) {
  let paramArray = []
  if (paramData && typeof paramData === 'object') {
    paramArray = Object.keys(paramData).map(k => `${k}=${paramData[k]}`)

addUrlSource (opt) {
  // opt.url 目标url
  // opt.param 参数对象
  let resultUrlPara = ''
  let resultUrl = opt.url
  let pageParamsStr = this.getParamStr(opt.param)
  let pageHasParam = Object.keys(opt.param).length
  // 活动页面有参数

    resultUrlPara = opt.url.indexOf('?') > -1 ? `&${pageParamsStr}` : `?${pageParamsStr}`

# 解析queryData

getParams () {
  let param = {}
  let queryStr = location.search || ''
  let value
  if (queryStr) {
    let queryStrArray = queryStr.replace('?', '').split('&') || []
    queryStrArray.length && queryStrArray.forEach(item => {
      let propertyObj = item.split('=')
      if (propertyObj[0]) {
        value = propertyObj[1] ? decodeURIComponent(propertyObj[1]) : propertyObj[1]
        param[propertyObj[0]] = value
      }
    })
  }
  return param
},

# 金额三位换算

numFormat(num) {//转换千分位
  if(!num){
    return '¥0'
  }
  if (num.toString().indexOf(".") !== -1) {
    var b = num.toLocaleString();
    return '¥' + b;
  } else {
    var c = num.toString().replace(/(\d)(?=(?:\d{3})+$)/g, "$1,");
    return '¥'+ c;
  }
},

# 防抖

const deboounce = (fn, delay)=> {
  var timer
  return function () {
    var context = this
    var args = arguments
    clearTimeout(timer)
   
    timer = setTimeout(function () {
      fn.apply(context, args)
    }, delay)
  }
};
